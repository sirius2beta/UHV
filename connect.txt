send AT+CPIN?
expect {
	"+CPIN: SIM PIN" goto unlock
	"+CPIN: READY" goto qcrmcall
	timeout 5 goto penic
}

unlock:
send AT+CPIN=2732
expect {
	"PB DONE"
	timeout 5 goto penic
}

qcrmcall:
send AT$QCRMCALL=1,1
expect {
	"ERROR" goto qcrmcall
	"NO CARRIER" goto qcrmcall
	"OK" goto continue
	timeout 5 goto panic
}

continue:
! killall -9 minicom


panic:
! killall -9 minicom
